ex4: Aluno (
	id int auto increment primary key,
	nome = varchar (100),
	data = date,
	desc_req = text	
);


Aluno_Necessidade_Especial (
    id_aluno INT NOT NULL,
    nec_especial VARCHAR(100) NOT NULL,
    PRIMARY KEY (id_aluno, necessidade_especial),
    FOREIGN KEY (id_aluno) REFERENCES Aluno(id)
);


Aluno_Area_Interesse (
    id_aluno INT NOT NULL,
    area_interesse VARCHAR(100) NOT NULL,
    PRIMARY KEY (id_aluno, area_interesse),
    FOREIGN KEY (id_aluno) REFERENCES aluno(id)
);

Aluno_Telefone (
    id_aluno INT NOT NULL,
    telefone VARCHAR(20) NOT NULL,
    PRIMARY KEY (id_aluno, telefone),
    FOREIGN KEY (id_aluno) REFERENCES aluno(id)
);

Aluno_Responsavel (
    id_aluno INT NOT NULL,
    nome_responsavel VARCHAR(100) NOT NULL,
    PRIMARY KEY (id_aluno, nome_responsavel),
    FOREIGN KEY (id_aluno) REFERENCES aluno(id)
);

Requerimento (
    id_requerimento INT AUTO_INCREMENT PRIMARY KEY,
    id_aluno INT NOT NULL,
    data_requerimento DATE NOT NULL,
    descricao_requerimento TEXT NOT NULL,
    FOREIGN KEY (id_aluno) REFERENCES aluno(id)
);
